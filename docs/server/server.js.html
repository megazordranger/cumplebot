server.js

require('custom-env').env();
const Telegraf = require('telegraf');
const schedule = require('node-schedule');

/**
 * Get DB models
 *
 * @type {Object}
 */
const models = require('./src/models');
/**
 * Get locale object for multi language support
 *
 * @type {Object}
 */
const locale = require('./src/locale');
/**
 * Get global functions
 *
 * @type {Object}
 */
const functions = require('./src/functions');
/**
 * Get global variables
 *
 * @type {Object}
 */
const { monthActions, dayActions } = require('./src/vars');
/**
 * Get bot fucntions
 *
 * @type {Object}
 */
const {
	getMonth,
	getDay,
	storeDate,
	birthdayMessage,
	start,
	help,
	callChannel,
} = require('./src/events');

/**
 * Get commands in all languages
 *
 * @type {Object}
 */
const { ADDME, CHANNEL, HELP } = functions.getCommands(locale);
const globals = [functions, locale, models];
const bot = new Telegraf(process.env.BOT_TOKEN);
const rule = new schedule.RecurrenceRule();
rule.hour = process.env.RULE_HOUR;

schedule.scheduleJob(rule.hour, birthdayMessage(...globals, bot));

bot.start(start(functions, locale));
bot.command(HELP, help(...globals));
bot.command(ADDME, getMonth(...globals));
bot.command(CHANNEL, callChannel(models, bot));
bot.action(monthActions, getDay(...globals));
bot.action(dayActions, storeDate(...globals));
bot.launch();
